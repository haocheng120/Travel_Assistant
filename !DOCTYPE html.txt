// 移除 PRE TRIP 標籤，純粹生成日期天數
function getTripDays() {
  const tabs = [];
  if (S.settings.start && S.settings.end) {
    const start = new Date(S.settings.start);
    const end = new Date(S.settings.end);
    let curr = new Date(start);
    let dayNum = 1;
    const wd = ['日','一','二','三','四','五','六'];
    while(curr <= end) {
      const yyyy = curr.getFullYear();
      const mm = String(curr.getMonth()+1).padStart(2, '0');
      const dd = String(curr.getDate()).padStart(2, '0');
      const id = `${yyyy}-${mm}-${dd}`;
      tabs.push({
        id: id,
        title: `DAY ${dayNum}`,
        val: `${curr.getMonth()+1}/${curr.getDate()} (${wd[curr.getDay()]})`
      });
      curr.setDate(curr.getDate() + 1);
      dayNum++;
    }
  }
  return tabs;
}

function generateDateTabs() {
  const tabsData = getTripDays();
  const tabContainer = document.getElementById('itinerary-tabs');
  const selectContainer = document.getElementById('ev-date-select');
  
  tabContainer.innerHTML = tabsData.map(d => `
    <div class="date-tab ${d.id === currentDayTab ? 'active' : ''}" onclick="switchDayTab('${d.id}')">
      <div class="tab-title">${d.title}</div>
      <div class="tab-val">${d.val}</div>
    </div>
  `).join('');
  
  selectContainer.innerHTML = tabsData.map(d => `<option value="${d.id}">${d.title} - ${d.val}</option>`).join('');
}

function switchDayTab(id) {
  if (currentDayTab === id) return;
  currentDayTab = id;
  generateDateTabs(); 
  renderItinerary();
  scrollToActiveTab();
}

function renderHome() {
  document.getElementById('trip-start').value = S.settings.start;
  document.getElementById('trip-end').value = S.settings.end;

  if(S.settings.start) {
    const start = new Date(S.settings.start);
    const now = new Date(); now.setHours(0,0,0,0);
    const diff = Math.ceil((start - now) / 86400000);
    document.getElementById('trip-date-display').textContent = `${S.settings.start} 啟程`;
    
    if(diff > 0) {
      document.getElementById('countdown-num').textContent = diff;
      document.getElementById('countdown-label').textContent = '天後出發';
    } else if (diff === 0) {
      document.getElementById('countdown-num').textContent = 'GO';
      document.getElementById('countdown-label').textContent = '就是今天！';
    } else {
      document.getElementById('countdown-num').textContent = Math.abs(diff);
      document.getElementById('countdown-label').textContent = '天前出發';
    }
  }
  
  renderChecklist();
}

function renderChecklist() {
  const container = document.getElementById('home-checklist');
  if (S.checklist.length === 0) {
    container.innerHTML = '<div style="color:var(--text-sub); font-size:12px; text-align:center; padding:10px;">目前清單是空的</div>';
    return;
  }
  container.innerHTML = S.checklist.map(item => `

      <input type="checkbox" class="checklist-checkbox" ${item.done ? 'checked' : ''} onchange="toggleCheckItem('${item.id}')">
      <span class="checklist-text ${item.done ? 'done' : ''}">${item.text}</span>
      <button class="checklist-del" onclick="deleteCheckItem('${item.id}')">✕</button>
    </div>
  `).join('');
}

function addCheckItem() {
  const input = document.getElementById('new-check-item');
  const text = input.value.trim();
  if (!text) return;
  S.checklist.push({ id: 'c' + Date.now(), text: text, done: false });
  input.value = ''; 
  saveData(); 
  renderChecklist();
}

function toggleCheckItem(id) {
  const item = S.checklist.find(i => i.id === id);
  if (item) { 
    item.done = !item.done; 
    saveData(); 
    renderChecklist(); 
  }
}

function deleteCheckItem(id) {
  S.checklist = S.checklist.filter(i => i.id !== id);
  saveData(); 
  renderChecklist();
}

  `).join('');

  if(sortableInstance) sortableInstance.destroy();
  sortableInstance = new Sortable(list, {

}

  const rateInput = document.getElementById('exp-rate-input').value;
  const budgetInput = document.getElementById('exp-budget-input').value;
  
  S.settings.rate = parseFloat(rateInput) || 46.3;

function renderExpense() {
  const rate = S.settings.rate || 46.3;
  const budget = S.settings.budget || 0;
  document.getElementById('exp-rate-input').value = rate;
  document.getElementById('exp-budget-input').value = budget;

  let totalTWD = 0;
  S.expenses.forEach(e => {
    totalTWD += e.currency === 'KRW' ? (e.amount / rate) : e.amount;
  });
  totalTWD = Math.round(totalTWD);
  
  document.getElementById('budget-title').textContent = `預算 NT$ ${budget.toLocaleString()}`;
  document.getElementById('budget-left').textContent = `剩餘 NT$ ${(budget - totalTWD).toLocaleString()}`;
  document.getElementById('total-twd').textContent = `NT$ ${totalTWD.toLocaleString()}`;
  document.getElementById('budget-fill').style.width = budget > 0 ? `${Math.min(100, (totalTWD / budget) * 100)}%` : '0%';

  const list = document.getElementById('expense-list');
  const empty = document.getElementById('expense-empty');
  
  if(!S.expenses.length) { list.innerHTML = ''; empty.style.display = 'block'; return; }
  empty.style.display = 'none';
  
  list.innerHTML = [...S.expenses].reverse().map(e => {
    const d_twd = e.currency === 'KRW' ? Math.round(e.amount / rate) : e.amount;
    const d_krw = e.currency === 'KRW' ? e.amount : Math.round(e.amount * rate);
    
    let taxHtml = '';
    if(e.taxStatus === '已退稅') taxHtml = `<span class="tax-badge tax-green">已退稅</span>`;
    else taxHtml = `<span class="tax-badge tax-orange">未退稅</span>`;

    return `
    <div class="exp-item">
      <div class="exp-info">
        <div class="exp-title">${e.name}</div>
        <div class="exp-meta">
          <span>${e.payMethod}</span>
          ${taxHtml}
        </div>
      </div>
      <div>
        <div class="exp-amount">NT$ ${d_twd.toLocaleString()}</div>
        <div style="font-size:11px; color:var(--text-sub); text-align:right;">(₩ ${d_krw.toLocaleString()})</div>
      </div>
    </div>`;
  }).join('');
}


// --- 觸控滑動邏輯 (已包含 Intervention 防護) ---
let startX = 0, startY = 0;
let itemToDelete = null;
let cardToReset = null;

document.addEventListener('touchstart', e => {
  if (e.touches.length > 1 || ['INPUT', 'SELECT', 'TEXTAREA', 'BUTTON', 'A'].includes(e.target.tagName)) return;
  startX = e.touches[0].clientX;
  startY = e.touches[0].clientY;
}, {passive: true});

document.addEventListener('touchmove', e => {
  if (!startX || !startY) return;
  let moveX = e.touches[0].clientX;
  let moveY = e.touches[0].clientY;
  
  // 如果判斷是左右滑動，且支援取消，則阻止預設捲動
  if (Math.abs(moveX - startX) > Math.abs(moveY - startY) && Math.abs(moveX - startX) > 10) {
    if (e.cancelable) {
      e.preventDefault(); 
    }
  }
}, { passive: false });

document.addEventListener('touchend', e => {
  if (!startX || !startY) return;
  let endX = e.changedTouches[0].clientX;
  let endY = e.changedTouches[0].clientY;
  let dx = endX - startX;
  let dy = endY - startY;

  if (Math.abs(dy) > Math.abs(dx) || Math.abs(dx) < 40) {
    startX = 0; startY = 0;
    return;
  }

  let wrapper = e.target.closest('.event-wrapper');
  if (wrapper && S.page === 'itinerary') {
    let card = wrapper.querySelector('.event-card');
    if (dx < -40) { 
      card.style.transform = 'translateX(-80px)';
      let eventId = wrapper.getAttribute('data-id');
      setTimeout(() => { showCustomConfirm(eventId, card); }, 150);
      startX = 0; startY = 0;
      return;
    } else if (dx > 40 && card.style.transform === 'translateX(-80px)') {
      card.style.transform = 'translateX(0px)';
      startX = 0; startY = 0;
      return; 
    }
  }

  let isItineraryDateArea = e.target.closest('#itinerary-date-area');

  if (dx < -40) {
    if (S.page === 'home') {
      goTo('itinerary');
    } else if (S.page === 'itinerary') {
      if (isItineraryDateArea) {
        const tabs = getTripDays();
        const idx = tabs.findIndex(t => t.id === currentDayTab);
        if (idx > -1 && idx < tabs.length - 1) {
          switchDayTab(tabs[idx+1].id);
        } else {
          goTo('expense');
        }
      } else {
        goTo('expense');
      }
    }
  } else if (dx > 40) {
    if (S.page === 'expense') {
      goTo('itinerary');
    } else if (S.page === 'itinerary') {
      if (isItineraryDateArea) {
        const tabs = getTripDays();
        const idx = tabs.findIndex(t => t.id === currentDayTab);
        if (idx > 0) {
          switchDayTab(tabs[idx-1].id);
        } else {
          goTo('home');
        }
      } else {
        goTo('home');
      }
    }
  }

  startX = 0; startY = 0;
});

function showCustomConfirm(id, cardNode) {
  itemToDelete = id;
  cardToReset = cardNode;
  document.getElementById('modal-confirm').classList.add('open');
}
function cancelDelete() {
  if (cardToReset) cardToReset.style.transform = 'translateX(0px)';
  document.getElementById('modal-confirm').classList.remove('open');
  itemToDelete = null; cardToReset = null;
}
function confirmDelete() {
  if(itemToDelete) {
    S.events = S.events.filter(e => e.id !== itemToDelete);
    saveData(); renderItinerary(); showToast('已刪除');
  }
  document.getElementById('modal-confirm').classList.remove('open');
  itemToDelete = null; cardToReset = null;
}

function saveEvent() {
  const name = document.getElementById('ev-name').value.trim();
  if(!name) { showToast('請填寫地點名稱'); return; }

  const targetDate = document.getElementById('ev-date-select').value;

  S.events.push({
    id: Date.now().toString(),
    date: targetDate,
    time: document.getElementById('ev-time').value,
    type: document.getElementById('ev-type').value,
    name: name,
    note: document.getElementById('ev-note').value
  });

  saveData();
  closeModal('modal-event');

  currentDayTab = targetDate;
  generateDateTabs();
  renderItinerary();
  setTimeout(scrollToActiveTab, 50);
}

function saveExpense() {
  const name = document.getElementById('exp-name').value.trim() || '未命名項目';
  const amount = parseFloat(document.getElementById('exp-amount').value);
  if(!amount) { showToast('請輸入金額'); return; }
  
  S.expenses.push({
    id: Date.now().toString(),
    name: name,
    payMethod: expPayMethod,
    taxStatus: expTaxStatus,
    currency: document.getElementById('exp-currency').value,
    amount: amount
  });
  saveData(); closeModal('modal-expense'); renderExpense();
}

function openMapModal() {
  const routeContainer = document.getElementById('map-route-list');
  
  if (S.events.length === 0) {
    routeContainer.innerHTML = '<div class="empty" style="padding: 40px 0;"><div class="empty-icon">🗺️</div><div>目前還沒有任何行程可以顯示喔！</div></div>';
    openModal('modal-map');
    return;
  }

  let sortedEvents = [...S.events].sort((a, b) => {
    if (a.date !== b.date) {
      return a.date.localeCompare(b.date); 
    }
    return a.time.localeCompare(b.time);   
  });

  let uniqueDates = [...new Set(sortedEvents.map(e => e.date))];
  
  const dayColors = ['#FF9AA2', '#FFDAC1', '#E2F0CB', '#B5EAD7', '#C7CEEA', '#E0BBE4', '#FFB7B2'];

  let html = '';
  let globalIndex = 1; 

  sortedEvents.forEach((ev, index) => {
    let dayIndex = uniqueDates.indexOf(ev.date);
    let nodeColor = dayColors[dayIndex % dayColors.length];
    
    let isLast = index === sortedEvents.length - 1;

    html += `
      <div style="display:flex; align-items:stretch; gap:16px;">
        <div style="display:flex; flex-direction:column; align-items:center; width: 32px;">
          <div style="width:32px; height:32px; border-radius:50%; background:${nodeColor}; color:#4A4643; display:flex; align-items:center; justify-content:center; font-weight:900; font-size:14px; z-index:2; box-shadow: 0 2px 6px rgba(0,0,0,0.1);">
            ${globalIndex}
          </div>
          ${!isLast ? `<div style="width:3px; flex:1; background:${nodeColor}; opacity:0.4; margin-top:4px; margin-bottom:4px; border-radius: 2px;"></div>` : `<div style="height: 16px;"></div>`}
        </div>
        
        <div style="flex:1; padding-bottom: 24px;">
          <div style="display:flex; align-items:center; gap: 8px; margin-bottom: 4px;">
            <span style="font-size:12px; font-weight:700; color:var(--text-sub); background: #EEEEEE; padding: 2px 8px; border-radius: 12px;">${ev.date}</span>
            <span style="font-size:13px; font-family:var(--font-serif); color:var(--primary-dark); font-weight:700;">${ev.time}</span>
          </div>
          <div style="font-size:16px; font-weight:700; color:var(--text-main); margin-bottom:8px;">${ev.name}</div>
          
          <a href="https://map.naver.com/v5/search/${encodeURIComponent(ev.name)}" target="_blank" style="display:inline-flex; align-items:center; gap:4px; font-size:12px; font-weight:700; color:#03c75a; text-decoration:none; border:1px solid rgba(3,199,90,0.3); background: rgba(3,199,90,0.05); padding:6px 12px; border-radius:6px; transition:all 0.2s;">
            🗺️ 尋找地圖位置
          </a>
        </div>
      </div>
    `;
    globalIndex++;
  });

  routeContainer.innerHTML = html;
  openModal('modal-map');
}
